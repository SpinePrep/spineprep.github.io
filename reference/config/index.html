    Example Configuration¶

    # Basic configuration
bids_dir: /path/to/bids
output_dir: /path/to/output
n_procs: 4

# Advanced configuration
motion:
  fd_threshold: 0.5
  dvars_threshold: 75

confounds:
  acompcor: true
  censor: true

registration:
  method: sct
  template: PAM50
